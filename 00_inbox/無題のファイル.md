## scaninit修正箇所

```
# -----------------------------
# 6. Full scan (BG)
# -----------------------------
echo -n "[?] Start background full port scan (-p-)? [y/N]: "
read -q "choice"  # -q は1文字だけ読み取って変数に格納
echo ""           # read -q は改行しないので手動で追加

if [[ "$choice" =~ ^[Yy]$ ]]; then
  echo "[*] Starting full port scan in background..."
  nohup nmap -Pn -n -p- "$TARGET_IP" \
    -oA "$OUT/nmap_full" \
    > "$LOG/nmap_full.log" 2>&1 &
  echo "[+] Background full scan started"
else
  echo "[*] Background full scan skipped"
fi
```

## fullcheck versionup

```
[ -n "$TMUX" ] && tmuxsync

if [ -z "$TARGET_IP" ] || [ -z "$OUT" ] || [ -z "$LOG" ]; then
  echo "[!] TARGET_IP / OUT / LOG is empty"
  echo "    Run: target <IP> [NAME]"
  return 1
fi

local full_base="$OUT/nmap_full"
local full_log="$LOG/nmap_full.log"
local full_nmap="${full_base}.nmap"
local init_nmap=""
local delta=""

if [ -n "$INIT_SCAN_SOURCE" ]; then
  # .nmap 拡張子が二重にならないよう、念のため除去してから付与
  init_nmap="${INIT_SCAN_SOURCE%.nmap}.nmap"
fi

# スキャン実行中かどうかの判定（pgrep）
if pgrep -af "nmap .* -p- .*${TARGET_IP}" >/dev/null 2>&1; then
  echo "[*] Full scan is still running for $TARGET_IP"
  echo "[*] Last 10 lines of progress:"
  if [ -f "$full_log" ]; then
    tail -n 10 "$full_log"
    echo "----------------------------------------"
    # nmapにシグナルを送って進捗(%)を表示させるテクニック（お好みで）
    pkill -HUP -f "nmap .* -p- .*${TARGET_IP}"
    sleep 0.5
  else
    echo "[!] Log file not found: $full_log"
  fi
  return 0
fi

if [ ! -f "$full_nmap" ]; then
  echo "[!] Full scan result not found: $full_nmap"
  return 1
fi

echo "[*] Full scan completed"
echo "----------------------------------------"
echo "[+] Open ports (full):"
grep -E '^[0-9]+/(tcp|udp)\s+open' "$full_nmap"

# 初期スキャンとの差分抽出
if [ -n "$init_nmap" ] && [ -f "$init_nmap" ]; then
  # commコマンドによる差分抽出
  delta=$(comm -13 \
    <(grep -E '^[0-9]+/(tcp|udp)\s+open' "$init_nmap" | cut -d/ -f1 | sort -un) \
    <(grep -E '^[0-9]+/(tcp|udp)\s+open' "$full_nmap" | cut -d/ -f1 | sort -un)
  )

  echo ""
  echo "[+] Delta (New ports found in full scan):"
  if [ -n "$delta" ]; then
    echo "$delta" | sed 's/^/  -> /' # 見やすく少しインデント
  else
    echo "  None (No new ports)"
  fi
else
  echo ""
  echo "[!] Initial scan source not found, skipping delta check"
fi
```
